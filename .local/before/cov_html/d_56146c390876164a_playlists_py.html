201 and starts playing that track

227 def add_new_playlist(self, tracks=[], name=None): 

228 """ 

229 Adds a new playlist to the list of playlists. If name is 

230 None or the name conflicts with an existing playlist, the 

231 user will be queried for a new name. 

232 

233 Returns the name of the new playlist, or None if it was 

234 not added. 

235 """ 

236 if name: 

237 if name in self.playlist_manager.playlists: 

238 name = dialogs.ask_for_playlist_name( 

239 self.get_panel().get_toplevel(), self.playlist_manager, name 

240 ) 

241 else: 

242 if tracks: 

243 artists = [] 

244 composers = [] 

245 albums = [] 

246 

247 for track in tracks: 

248 artist = track.get_tag_display('artist', artist_compilations=False) 

249 

250 if artist is not None: 

251 artists += [artist] 

252 

253 composer = track.get_tag_display( 

254 'composer', artist_compilations=False 

255 ) 

256 

257 if composer is not None: 

258 composers += composer 

259 

260 album = track.get_tag_display('album') 

261 

262 if album is not None: 

263 albums += album 

264 

265 artists = list(set(artists))[:3] 

266 composers = list(set(composers))[:3] 

267 albums = list(set(albums))[:3] 

268 

269 if len(artists) > 0: 

270 name = artists[0] 

271 

272 if len(artists) > 2: 

273 # TRANSLATORS: Playlist title suggestion with more 

274 # than two values 

275 name = _('%(first)s, %(second)s and others') % { 

276 'first': artists[0], 

277 'second': artists[1], 

278 } 

279 elif len(artists) > 1: 

280 # TRANSLATORS: Playlist title suggestion with two values 

281 name = _('%(first)s and %(second)s') % { 

282 'first': artists[0], 

283 'second': artists[1], 

284 } 

285 elif len(composers) > 0: 

286 name = composers[0] 

287 

288 if len(composers) > 2: 

289 # TRANSLATORS: Playlist title suggestion with more 

290 # than two values 

291 name = _('%(first)s, %(second)s and others') % { 

292 'first': composers[0], 

293 'second': composers[1], 

294 } 

295 elif len(composers) > 1: 

296 # TRANSLATORS: Playlist title suggestion with two values 

297 name = _('%(first)s and %(second)s') % { 

298 'first': composers[0], 

299 'second': composers[1], 

300 } 

301 elif len(albums) > 0: 

302 name = albums[0] 

303 

304 if len(albums) > 2: 

305 # TRANSLATORS: Playlist title suggestion with more 

306 # than two values 

307 name = _('%(first)s, %(second)s and others') % { 

308 'first': albums[0], 

309 'second': albums[1], 

310 } 

311 elif len(albums) > 1: 

312 # TRANSLATORS: Playlist title suggestion with two values 

313 name = _('%(first)s and %(second)s') % { 

314 'first': albums[0], 

315 'second': albums[1], 

316 } 

317 else: 

318 name = '' 

319 

320 name = dialogs.ask_for_playlist_name( 

321 self.get_panel().get_toplevel(), self.playlist_manager, name 

322 ) 

323 

324 if name is not None: 

325 # Create the playlist from all of the tracks 

326 new_playlist = Playlist(name) 

327 new_playlist.extend(tracks) 

328 # We are adding a completely new playlist with tracks so we save it 

329 self.playlist_manager.save_playlist(new_playlist) 

330 

331 return name

333 def _load_playlist_nodes(self, playlist): 

334 """ 

335 Loads the playlist tracks into the node for the specified playlist 

336 """ 

337 if playlist not in self.playlist_nodes: 

338 return 

339 

340 expanded = self.tree.row_expanded( 

341 self.model.get_path(self.playlist_nodes[playlist]) 

342 ) 

343 

344 self._clear_node(self.playlist_nodes[playlist]) 

345 parent = self.playlist_nodes[playlist] 

346 for track in playlist: 

347 if not track: 

348 continue 

349 wrapper = TrackWrapper(track, playlist) 

350 row = (self.track_image, str(wrapper), wrapper) 

351 self.model.append(parent, row) 

352 

353 if expanded: 

354 self.tree.expand_row( 

355 self.model.get_path(self.playlist_nodes[playlist]), False 

356 )

506 def _on_playlist_added(self, type, object, playlist_name): 

507 new_playlist = self.playlist_manager.get_playlist(playlist_name) 

508 

509 for oldpl in self.playlist_nodes: 

510 if oldpl.name == playlist_name: # Name already exists 

511 if oldpl is not new_playlist: 

512 node = self.playlist_nodes[oldpl] 

513 # Replace the playlist object in {playlist: iter} cache. 

514 del self.playlist_nodes[oldpl] 

515 self.playlist_nodes[new_playlist] = node 

516 # Replace the playlist object in tree model. 

517 self.model[node][2] = new_playlist 

518 break 

519 else: # Name doesn't exist yet 

520 self.playlist_nodes[new_playlist] = self.model.append( 

521 self.custom, [self.playlist_image, playlist_name, new_playlist] 

522 ) 

523 self.tree.expand_row(self.model.get_path(self.custom), False)

599 def drag_data_received(self, tv, context, x, y, selection, info, etime): 

600 """ 

601 Called when someone drags some thing onto the playlist panel 

602 """ 

603 if info == self.playlist_name_info: 

604 # We are being dragged a playlist so 

605 # we have to reorder them 

606 playlist_name = selection.get_text() 

607 drag_source = self.tree.get_selected_page() 

608 # verify names 

609 if drag_source is not None: 

610 if drag_source.name == playlist_name: 

611 drop_info = tv.get_dest_row_at_pos(x, y) 

612 drag_source_iter = self.playlist_nodes[drag_source] 

613 if drop_info: 

614 path, position = drop_info 

615 drop_target_iter = self.model.get_iter(path) 

616 drop_target = self.model.get_value(drop_target_iter, 2) 

617 if position == Gtk.TreeViewDropPosition.BEFORE: 

618 # Put the playlist before drop_target 

619 self.model.move_before(drag_source_iter, drop_target_iter) 

620 self.playlist_manager.move( 

621 playlist_name, drop_target.name, after=False 

622 ) 

623 else: 

624 # put the playlist after drop_target 

625 self.model.move_after(drag_source_iter, drop_target_iter) 

626 self.playlist_manager.move( 

627 playlist_name, drop_target.name, after=True 

628 ) 

629 # Even though we are doing a move we still don't 

630 # call the delete method because we take care 

631 # of it above by moving instead of inserting/deleting 

632 context.finish(True, False, etime) 

633 else: 

634 self._drag_data_received_uris(tv, context, x, y, selection, info, etime)

636 def _drag_data_received_uris(self, tv, context, x, y, selection, info, etime): 

637 """ 

638 Called by drag_data_received when the user drags URIs onto us 

639 """ 

640 locs = list(selection.get_uris()) 

641 drop_info = tv.get_dest_row_at_pos(x, y) 

642 if drop_info: 

643 path, position = drop_info 

644 iter = self.model.get_iter(path) 

645 drop_target = self.model.get_value(iter, 2) 

646 

647 # if the current item is a track, use the parent playlist 

648 insert_index = None 

649 if isinstance(drop_target, TrackWrapper): 

650 current_playlist = drop_target.playlist 

651 drop_target_index = current_playlist.index(drop_target.track) 

652 # Adjust insert position based on drop position 

653 if ( 

654 position == Gtk.TreeViewDropPosition.BEFORE 

655 or position == Gtk.TreeViewDropPosition.INTO_OR_BEFORE 

656 ): 

657 # By default adding tracks inserts it before so we do not 

658 # have to modify the insert index 

659 insert_index = drop_target_index 

660 else: 

661 # If we want to go after we have to append 1 

662 insert_index = drop_target_index + 1 

663 else: 

664 current_playlist = drop_target 

665 

666 # Since the playlist do not have very good support for 

667 # duplicate tracks we have to perform some trickery 

668 # to make this work properly in all cases 

669 try: 

670 remove_track_index = current_playlist.index( 

671 self.tree.get_selected_track() 

672 ) 

673 except ValueError: 

674 remove_track_index = None 

675 if insert_index is not None and remove_track_index is not None: 

676 # Since remove_track_index will be removed before 

677 # the new track is inserted we have to offset the 

678 # insert index 

679 if insert_index > remove_track_index: 

680 insert_index = insert_index - 1 

681 

682 # Delete the track before adding the other one 

683 # so we do not get duplicates 

684 # right now the playlist does not support 

685 # duplicate tracks very well 

686 if context.get_selected_action() == Gdk.DragAction.MOVE: 

687 # On a move action the second True makes the 

688 # drag_data_delete function called 

689 context.finish(True, True, etime) 

690 else: 

691 context.finish(True, False, etime) 

692 

693 # Add the tracks we found to the internal playlist 

694 # TODO: have it pass in existing tracks? 

695 (tracks, playlists) = self.tree.get_drag_data(locs) 

696 

697 if insert_index is not None: 

698 current_playlist[insert_index:insert_index] = tracks 

699 else: 

700 current_playlist.extend(tracks) 

701 

702 self._load_playlist_nodes(current_playlist) 

703 

704 # Do we save in the case when a user drags a file onto a playlist 

705 # in the playlist panel? note that the playlist does not have to 

706 # be open for this to happen 

707 self.playlist_manager.save_playlist(current_playlist, overwrite=True) 

708 else: 

709 # If the user dragged files prompt for a new playlist name 

710 # else if they dragged a playlist add the playlist 

711 

712 # We don't want the tracks in the playlists to be added to the 

713 # master tracks list so we pass in False 

714 (tracks, playlists) = self.tree.get_drag_data(locs, False) 

715 # First see if they dragged any playlist files 

716 for new_playlist in playlists: 

717 # We are adding a completely new playlist with tracks so 

718 # we save it. This will trigger playlist_added. 

719 self.playlist_manager.save_playlist(new_playlist, overwrite=True) 

720 

721 # After processing playlist proceed to ask the user for the 

722 # name of the new playlist to add and add the tracks to it 

723 if len(tracks) > 0: 

724 self.add_new_playlist(tracks)

726 def drag_data_delete(self, tv, context): 

727 """ 

728 Called after a drag data operation is complete 

729 and we want to delete the source data 

730 """ 

731 if Gdk.drag_drop_succeeded(context): 

732 self.remove_selected_track()

734 def drag_get_data(self, tv, context, selection_data, info, time): 

735 """ 

736 Called when someone drags something from the playlist 

737 """ 

738 # TODO based on info determine what we set in selection_data 

739 if info == self.playlist_name_info: 

740 pl = self.tree.get_selected_page() 

741 if pl is not None: 

742 selection_data.set_text(pl.name, len(pl.name)) 

743 else: 

744 pl = self.tree.get_selected_page() 

745 if pl is not None: 

746 tracks = pl[:] 

747 else: 

748 tracks = self.tree.get_selected_tracks() 

749 

750 if not tracks: 

751 return 

752 

753 for track in tracks: 

754 DragTreeView.dragged_data[track.get_loc_for_io()] = track 

755 

756 uris = trax.util.get_uris_from_tracks(tracks) 

757 selection_data.set_uris(uris)

759 def drag_motion(self, tv, context, x, y, time): 

760 """ 

761 Sets the appropriate drag action based on what we are hovering over 

762 

763 hovering over playlists causes the copy action to occur 

764 hovering over tracks within the same playlist causes the move 

765 action to occur 

766 hovering over tracks within different playlist causes the move 

767 action to occur 

768 

769 Called on the destination widget 

770 """ 

771 # Reset any target to be default to moving tracks 

772 self.tree.enable_model_drag_dest([self.track_target], Gdk.DragAction.DEFAULT) 

773 # Determine where the drag is coming from 

774 dragging_playlist = False 

775 if tv == self.tree: 

776 selected_playlist = self.tree.get_selected_page() 

777 if selected_playlist is not None: 

778 dragging_playlist = True 

779 

780 # Find out where they are dropping onto 

781 drop_info = tv.get_dest_row_at_pos(x, y) 

782 if drop_info: 

783 path, position = drop_info 

784 iter = self.model.get_iter(path) 

785 drop_target = self.model.get_value(iter, 2) 

786 

787 if isinstance(drop_target, Playlist): 

788 if dragging_playlist: 

789 # If we drag onto we copy, if we drag between we move 

790 if ( 

791 position == Gtk.TreeViewDropPosition.INTO_OR_BEFORE 

792 or position == Gtk.TreeViewDropPosition.INTO_OR_AFTER 

793 ): 

794 Gdk.drag_status(context, Gdk.DragAction.COPY, time) 

795 else: 

796 Gdk.drag_status(context, Gdk.DragAction.MOVE, time) 

797 # Change target as well 

798 self.tree.enable_model_drag_dest( 

799 [self.playlist_target], Gdk.DragAction.DEFAULT 

800 ) 

801 else: 

802 Gdk.drag_status(context, Gdk.DragAction.COPY, time) 

803 elif isinstance(drop_target, TrackWrapper): 

804 # We are dragging onto another track 

805 # make it a move operation if we are only dragging 

806 # tracks within our widget 

807 # We do a copy if we are draggin from another playlist 

808 if Gtk.drag_get_source_widget(context) == tv and not dragging_playlist: 

809 Gdk.drag_status(context, Gdk.DragAction.MOVE, time) 

810 else: 

811 Gdk.drag_status(context, Gdk.DragAction.COPY, time) 

812 else: 

813 # Prevent drop operation by changing the targets 

814 self.tree.enable_model_drag_dest( 

815 self.deny_targets, Gdk.DragAction.DEFAULT 

816 ) 

817 return False 

818 return True 

819 else: # No drop info 

820 if dragging_playlist: 

821 Gdk.drag_status(context, Gdk.DragAction.MOVE, time) 

822 # Change target as well 

823 self.tree.enable_model_drag_dest( 

824 [self.playlist_target], Gdk.DragAction.DEFAULT 

825 ) 

826 return True 

827 return False

829 def on_key_released(self, widget, event): 

830 """ 

831 Called when a key is released in the tree 

832 """ 

833 if event.keyval == Gdk.KEY_Menu: 

834 (mods, paths) = self.tree.get_selection().get_selected_rows() 

835 if paths and paths[0]: 

836 iter = self.model.get_iter(paths[0]) 

837 pl = self.model.get_value(iter, 2) 

838 # Based on what is selected determines what 

839 # menu we will show 

840 if isinstance(pl, Playlist): 

841 Gtk.Menu.popup( 

842 self.playlist_menu, None, None, None, None, 0, event.time 

843 ) 

844 elif isinstance(pl, SmartPlaylist): 

845 Gtk.Menu.popup( 

846 self.smart_menu, None, None, None, None, 0, event.time 

847 ) 

848 elif isinstance(pl, TrackWrapper): 

849 Gtk.Menu.popup( 

850 self.track_menu, None, None, None, None, 0, event.time 

851 ) 

852 else: 

853 Gtk.Menu.popup( 

854 self.default_menu, None, None, None, None, 0, event.time 

855 ) 

856 return True 

857 

858 if event.keyval == Gdk.KEY_Left: 

859 (mods, paths) = self.tree.get_selection().get_selected_rows() 

860 if paths and paths[0]: 

861 self.tree.collapse_row(paths[0]) 

862 return True 

863 

864 if event.keyval == Gdk.KEY_Right: 

865 (mods, paths) = self.tree.get_selection().get_selected_rows() 

866 if paths and paths[0]: 

867 self.tree.expand_row(paths[0], False) 

868 return True 

869 

870 if event.keyval == Gdk.KEY_Delete: 

871 (mods, paths) = self.tree.get_selection().get_selected_rows() 

872 if paths and paths[0]: 

873 iter = self.model.get_iter(paths[0]) 

874 pl = self.model.get_value(iter, 2) 

875 # Based on what is selected determines what 

876 # menu we will show 

877 if isinstance(pl, (Playlist, SmartPlaylist)): 

878 self.remove_playlist(pl) 

879 elif isinstance(pl, TrackWrapper): 

880 self.remove_selected_track() 

881 return True 

882 return False
