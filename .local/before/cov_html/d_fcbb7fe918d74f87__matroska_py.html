59class Ebml: 

60 """EBML parser. 

61 

62 Usage: Ebml(location, tags).parse() 

63 where `tags` is a dictionary of the form {id: (name, type)}. 

64 """

66 ## Constructor and destructor 

67

70 self.open(location)

72 def __del__(self): 

73 self.close()

75 ## File access. 

76 ## These can be overridden to provide network support.

78 def open(self, location): 

79 """Open a location and set self.size.""" 

80 self.file = f = open(location, 'rb') 

81 f = self.file 

82 f.seek(0, 2) 

83 self.size = f.tell() 

84 f.seek(0, 0) 

85 

86 def seek(self, offset, mode): 

87 self.file.seek(offset, mode) 

88 

89 def tell(self): 

90 return self.file.tell() 

91 

92 def read(self, length): 

93 return self.file.read(length) 

94 

95 def close(self): 

96 self.file.close()

98 ## Element reading 

99 

100 def readID(self): 

101 b = self.read(1) 

102 b1 = ord(b) 

103 if b1 & 0b10000000: # 1 byte 

104 return b 

105 elif b1 & 0b01000000: # 2 bytes 

106 return unpack(">H", b + self.read(1))[0] 

107 elif b1 & 0b00100000: # 3 bytes 

108 return unpack(">L", b"\0" + b + self.read(2))[0] 

109 elif b1 & 0b00010000: # 4 bytes 

110 return unpack(">L", b + self.read(3))[0] 

111 else: 

112 raise EbmlException("invalid element ID (leading byte 0x%02X)" % b1) 

113 

114 def readSize(self): 

115 b1 = ord(self.read(1)) 

116 if b1 & 0b10000000: # 1 byte 

117 return b1 & 0b01111111 

118 elif b1 & 0b01000000: # 2 bytes 

119 return unpack(">H", bchr(b1 & 0b00111111) + self.read(1))[0] 

120 elif b1 & 0b00100000: # 3 bytes 

121 return unpack(">L", b"\0" + bchr(b1 & 0b00011111) + self.read(2))[0] 

122 elif b1 & 0b00010000: # 4 bytes 

123 return unpack(">L", bchr(b1 & 0b00001111) + self.read(3))[0] 

124 elif b1 & 0x00001000: # 5 bytes 

125 return unpack(">Q", b"\0\0\0" + bchr(b1 & 0b00000111) + self.read(4))[0] 

126 elif b1 & 0b00000100: # 6 bytes 

127 return unpack(">Q", b"\0\0" + bchr(b1 & 0b0000011) + self.read(5))[0] 

128 elif b1 & 0b00000010: # 7 bytes 

129 return unpack(">Q", b"\0" + bchr(b1 & 0b00000001) + self.read(6))[0] 

130 elif b1 & 0b00000001: # 8 bytes 

131 return unpack(">Q", b"\0" + self.read(7))[0] 

132 else: 

133 assert b1 == 0 

134 raise EbmlException("undefined element size") 

135 

136 def readInteger(self, length, signed): 

137 if length == 1: 

138 value = ord(self.read(1)) 

139 elif length == 2: 

140 value = unpack(">H", self.read(2))[0] 

141 elif length == 3: 

142 value = unpack(">L", b"\0" + self.read(3))[0] 

143 elif length == 4: 

144 value = unpack(">L", self.read(4))[0] 

145 elif length == 5: 

146 value = unpack(">Q", b"\0\0\0" + self.read(5))[0] 

147 elif length == 6: 

148 value = unpack(">Q", b"\0\0" + self.read(6))[0] 

149 elif length == 7: 

150 value = unpack(">Q", b"\0" + (self.read(7)))[0] 

151 elif length == 8: 

152 value = unpack(">Q", self.read(8))[0] 

153 else: 

154 raise EbmlException("don't know how to read %r-byte integer" % length) 

155 if signed: 

156 nbits = (8 - length) + 8 * (length - 1) 

157 if value >= (1 << (nbits - 1)): 

158 value -= 1 << nbits 

159 return value 

160 

161 def readFloat(self, length): 

162 if length == 4: 

163 return unpack('>f', self.read(4))[0] 

164 elif length == 8: 

165 return unpack('>d', self.read(8))[0] 

166 else: 

167 raise EbmlException("don't know how to read %r-byte float" % length) 

168 

169 ## Parsing 

170 

171 def parse(self, from_=0, to=None): 

172 """Parses EBML from `from_` (inclusive) to `to` (exclusive). 

173 

174 Note that not all streams support seeking backwards, so prepare to handle 

175 an exception if you try to parse from arbitrary position. 

176 """ 

177 if to is None: 

178 to = self.size 

179 self.seek(from_, 0) 

180 node = {} 

181 # Iterate over current node's children. 

182 while self.tell() < to: 

183 try: 

184 id = self.readID() 

185 except EbmlException as e: 

186 # Invalid EBML header. We can't reliably get any more data from 

187 # this level, so just return anything we have. 

188 warn(EbmlWarning(e)) 

189 return node 

190 size = self.readSize() 

191 if size == 0b01111111: 

192 warn(EbmlWarning("don't know how to handle unknown-sized element")) 

193 size = to - self.tell() 

194 try: 

195 key, type_ = self.tags[id] 

196 except KeyError: 

197 self.seek(size, 1) 

198 continue 

199 try: 

200 if type_ is SINT: 

201 value = self.readInteger(size, True) 

202 elif type_ is UINT: 

203 value = self.readInteger(size, False) 

204 elif type_ is FLOAT: 

205 value = self.readFloat(size) 

206 elif type_ is STRING: 

207 value = self.read(size).decode('ascii') 

208 elif type_ is UTF8: 

209 value = self.read(size).decode('utf-8') 

210 elif type_ is DATE: 

211 us = self.readInteger(size, True) / 1000.0 # ns to us 

212 from datetime import datetime, timedelta 

213 

214 value = datetime(2001, 1, 1) + timedelta(microseconds=us) 

215 elif type_ is MASTER: 

216 tell = self.tell() 

217 value = self.parse(tell, tell + size) 

218 elif type_ is BINARY: 

219 value = BinaryData(self.read(size)) 

220 else: 

221 assert False, type_ 

222 except (EbmlException, UnicodeDecodeError) as e: 

223 warn(EbmlWarning(e)) 

224 else: 

225 try: 

226 parentval = node[key] 

227 except KeyError: 

228 parentval = node[key] = [] 

229 parentval.append(value) 

230 return node

238class GioEbml(Ebml): 

239 # NOTE: All seeks are faked using InputStream.skip because we need to use 

240 # BufferedInputStream but it does not implement Seekable. 

241 

242 def open(self, location): 

243 f = Gio.File.new_for_commandline_arg(location) 

244 self.buffer = Gio.BufferedInputStream.new(f.read()) 

245 self._tell = 0 

246 

247 self.size = f.query_info( 

248 'standard::size', Gio.FileQueryInfoFlags.NONE, None 

249 ).get_size() 

250 

251 def seek(self, offset, mode): 

252 if mode == 0: 

253 skip = offset - self._tell 

254 elif mode == 1: 

255 skip = offset 

256 elif mode == 2: 

257 skip = self.size - self._tell + offset 

258 else: 

259 raise ValueError("invalid seek mode: %r" % mode) 

260 if skip < 0: 

261 raise NotImplementedError("seeking backwards not supported") 

262 self._tell += skip 

263 self.buffer.skip(skip) 

264 

265 def tell(self): 

266 return self._tell 

267 

268 def read(self, length): 

269 result = self.buffer.read_bytes(length).get_data() 

270 self._tell += len(result) 

271 return result 

272 

273 def close(self): 

274 self.buffer.close()

381def gio_location(location): 

382 """Convert location to GIO-compatible location. 

383 

384 This works around broken behaviour in the Win32 GIO port (it converts paths 

385 into UTF-8 and requires them to be specified in UTF-8 as well). 

386 

387 :type location: str 

388 :rtype: bytes 

389 """ 

390 if sys.platform == 'win32' and '://' not in location: 

391 if isinstance(location, bytes): 

392 # Decode the path according to the FS encoding to get the Unicode 

393 # representation first. If the path is in a different encoding, 

394 # this step will fail. 

395 location = location.decode(sys.getfilesystemencoding()) 

396 location = location.encode('utf-8') 

397 return location
