85class BrowserPage(WebKit2.WebView): 

86 def __init__(self, builder, user_agent): 

87 WebKit2.WebView.__init__(self) 

88 

89 self.hometrack = None 

90 self.__user_agent = user_agent 

91 

92 builder.connect_signals(self) 

93 event.add_callback(self.on_playback_start, 'playback_track_start') 

94 

95 def destroy(self): 

96 event.remove_callback(self.on_playback_start, 'playback_track_start') 

97 

98 def on_playback_start(self, type, player, track): 

99 self.hometrack = track 

100 self.load_wikipedia_page(track) 

101 

102 def on_home_button_clicked(self, button): 

103 if self.hometrack is not None: 

104 self.load_wikipedia_page(self.hometrack) 

105 

106 def on_refresh_button_clicked(self, button): 

107 self.reload() 

108 

109 def on_back_button_clicked(self, button): 

110 self.go_back() 

111 

112 def on_forward_button_clicked(self, button): 

113 self.go_forward() 

114 

115 @common.threaded 

116 def load_wikipedia_page(self, track): 

117 if track != self.hometrack: 

118 return 

119 

120 artist = track.get_tag_display('artist') 

121 language = settings.get_option('plugin/wikipedia/language', 'en') 

122 if language not in LANGUAGES: 

123 log.error('Provided language "%s" not found.' % language) 

124 language = 'en' 

125 

126 artist = urllib.parse.quote(artist.encode('utf-8'), '') 

127 url = "https://%s.m.wikipedia.org/wiki/Special:Search/%s" % (language, artist) 

128 

129 try: 

130 html = common.get_url_contents(url, self.__user_agent) 

131 if not isinstance(html, str): 

132 html = html.decode("utf-8") 

133 except urllib.error.URLError as e: 

134 log.error(e) 

135 log.error( 

136 "Error occurred when trying to retrieve Wikipedia page " 

137 "for %s." % artist 

138 ) 

139 html = ( 

140 """ 

141 <p style="color: red">No Wikipedia page found for <strong>%s</strong></p> 

142 """ 

143 % artist 

144 ) 

145 

146 GLib.idle_add(self.load_html, html, url)
